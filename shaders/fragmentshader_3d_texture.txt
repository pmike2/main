#version 410 core

uniform sampler3D texture_3d;
uniform sampler2D index_2d;
uniform screen_with;
uniform screen_height;

in vec3 tex_coord;
out vec4 color;

void main() {
	//color= texture(texture_3d, tex_coord);
	vec2 uv = vec2(gl_FragCoord.x/ screen_with, gl_FragCoord.y/ screen_height);
	vec3 i = texture(index_2d, uv);
	color= texture(texture_3d, i);
}
