#version 410

in vec2 tex_coords;
out vec4 color;

uniform sampler2D text;
uniform vec3 text_color;
uniform float alpha;

void main() {
	vec4 sampled= vec4(1.0, 1.0, 1.0, texture(text, tex_coords).r);
	color= vec4(text_color, alpha)* sampled;
	//color= vec4(1.0, 0.0, 0.0, 1.0);
	//color= vec4(text_color, 1.0);
}
