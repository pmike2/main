#version 410


uniform mat4 camera2clip_matrix;

in vec2 position_in;
in vec2 tex_coord_in;
in float current_layer_in;
in float hit_in;
in float alpha_in;

out vec2 tex_coord;
out float current_layer;
out float hit;
out float alpha;


void main() {
	gl_Position= camera2clip_matrix* vec4(position_in.rg, 0.0, 1.0);
	tex_coord= tex_coord_in;
	current_layer= current_layer_in;
	hit= hit_in;
	alpha= alpha_in;
}
