#version 410 core


uniform sampler2DArray texture_array;


in vec2 tex_coord;
in float current_layer;
in vec4 vertex_color;


out vec4 frag_color;


void main(void) {
	float alpha= texture(texture_array, vec3(tex_coord.x, tex_coord.y, current_layer)).r;
	frag_color= vec4(vertex_color.rgb, alpha);
}
